//------------------------------------------------------------ // 月次発電量の計算（積算発電量の差分方式） //------------------------------------------------------------ function calculateMonthly(dailyData) { if (!dailyData || dailyData.length < 2) return {}; // 日付順にソート dailyData.sort((a, b) => new Date(a.date) - new Date(b.date)); const dailyEnergy = []; for (let i = 1; i < dailyData.length; i++) { const today = dailyData[i]; const yesterday = dailyData[i - 1]; // 積算発電量の差分（Wh） const diffWh = today.totalWh - yesterday.totalWh; // 異常値（負値）は 0 とみなす const diffKWh = diffWh > 0 ? diffWh / 1000 : 0; dailyEnergy.push({ date: today.date, kwh: diffKWh }); } // 月次集計 const monthly = {}; dailyEnergy.forEach(d => { const ym = d.date.slice(0, 7); // "YYYY-MM" if (!monthly[ym]) { monthly[ym] = { totalKwh: 0, days: 0 }; } monthly[ym].totalKwh += d.kwh; monthly[ym].days += 1; }); return monthly; } //------------------------------------------------------------ // 月次サマリー表の描画 //------------------------------------------------------------ function renderMonthlyTable(monthly) { const tbody = document.getElementById("monthly-body"); tbody.innerHTML = ""; const months = Object.keys(monthly).sort(); months.forEach(ym => { const m = monthly[ym]; const avgKw = m.totalKwh / (m.days * 24); // 平均出力(kW) const tr = document.createElement("tr"); tr.innerHTML = ` ${ym} ${m.totalKwh.toFixed(2)} ${avgKw.toFixed(2)} ${m.days} `; tbody.appendChild(tr); }); document.getElementById("monthly-count").textContent = months.length + "件"; // データがない場合の表示切替 document.getElementById("monthly-empty").style.display = months.length === 0 ? "block" : "none"; } //------------------------------------------------------------ // 月次チャートの描画（Chart.js） //------------------------------------------------------------ function renderMonthlyChart(monthly) { const labels = Object.keys(monthly).sort(); const data = labels.map(ym => monthly[ym].totalKwh); const ctx = document.getElementById("monthlyChart"); new Chart(ctx, { type: "bar", data: { labels, datasets: [{ label: "月次発電量 (kWh)", data, backgroundColor: "#4e79a7" }] }, options: { responsive: true, scales: { y: { beginAtZero: true } } } }); }
